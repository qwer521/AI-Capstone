class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        unordered_map<int, int> m;
        for(int i = 0; i < nums.size(); i++){
            for(int j = i + 1; j <n ums.size(); j++){
                m[nums[i] * nums[j]]++;
            }
        }
        int ans = 0;
        for(auto i:m){
            if(i.second > 1) ans += 4 * i.second * (i.second - 1);
        }
        return ans;
    }
};
